ENTITY padstack_definition
  SUPERTYPE OF(ONEOF(
       design_layer_type_specific_padstack_definition,
       stratum_type_independent_padstack_definition,
       stratum_occurrence_specific_padstack_definition))
  SUBTYPE OF (ee_product_definition);
  reference_implementation : reference_packaged_part_interconnect_implementation;
  teardrop_taper_start_distance : OPTIONAL length_data_element;
  teardrop_taper_end_distance : OPTIONAL length_data_element;
  snowball_start_distance : OPTIONAL length_data_element;
  snowball_end_distance : OPTIONAL length_data_element;
  snowball_initial_spacing : OPTIONAL length_data_element;  
  snowball_end_spacing : OPTIONAL length_data_element;  
  snowball_initial_radius : OPTIONAL length_data_element;  
  snowball_end_radius : OPTIONAL length_data_element;  
  snowball_quantity_per_trace : OPTIONAL positive_integer;  
INVERSE
  shape : padstack_definition_shape FOR shape_characterized_padstack_definition;
WHERE
 WR1: SIZEOF (QUERY( prpc <* 
   USEDIN(SELF\ee_product_definition.version.versioned_item,
       'AP210_ARM.EE_PRODUCT_RELATED_PRODUCT_CATEGORY.OF_PRODUCT') | 
              (prpc.name = 'template model')
             )) >= 0;
END_ENTITY; -- padstack_definition
