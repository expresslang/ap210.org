  ENTITY printed_component_link
    SUBTYPE OF (managed_design_object);

    INVERSE

      WR2 : subsequent_printed_component\assembly_component.usage_definition = definition.subsequent_printed_part_template;
      WR3 : precedent_printed_component <> subsequent_printed_component;
  END_ENTITY; -- printed_component_link
