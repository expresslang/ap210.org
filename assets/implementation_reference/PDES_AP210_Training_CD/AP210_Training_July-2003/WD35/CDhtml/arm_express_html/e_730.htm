ENTITY printed_part_template

  ENTITY printed_part_template
    SUPERTYPE OF (ONEOF (layout_macro_floor_plan_template,
 printed_connector_template))
    SUBTYPE OF (part_template);
      of_stratum_technology       : OPTIONAL stratum_technology;
      material_stack              : OPTIONAL SET[1:?] OF stratum_technology_link;
      printed_part_template_stack : OPTIONAL SET[1:?] OF printed_part_template_link;
      implemented_function        : functional_unit_usage_view;
    DERIVE
      access_mechanisms : SET[1:?] OF printed_part_template_terminal := raw_access_mechanisms + printed_part_template_access_mechanisms(printed_part_template_stack);
    INVERSE
      raw_access_mechanisms : SET[0:?] OF printed_part_template_terminal FOR associated_definition;
    WHERE
      WR1 : ((EXISTS(of_stratum_technology) XOR EXISTS(material_stack)) AND (EXISTS(of_stratum_technology) XOR EXISTS(printed_part_template_stack)) AND (EXISTS(material_stack) XOR EXISTS(printed_part_template_stack))) OR ('AP210_ARM.LAYOUT_MACRO_FLOOR_PLAN_TEMPLATE' IN TYPEOF(SELF));
      WR2 : (NOT (EXISTS(material_stack)) OR (SIZEOF(QUERY(temp <* material_stack | EXISTS(temp.precedent_stratum_technology_link))) = SIZEOF(material_stack) - 1)) OR ('AP210_ARM.LAYOUT_MACRO_FLOOR_PLAN_TEMPLATE' IN TYPEOF(SELF));
      WR3 : (SIZEOF(QUERY(prpc <* USEDIN(associated_definition\ee_product_definition.version.versioned_item,
 'AP210_ARM.EE_PRODUCT_RELATED_PRODUCT_CATEGORY.OF_PRODUCT') | (prpc.name = 'printed part'))) >= 0) OR ('AP210_ARM.LAYOUT_MACRO_FLOOR_PLAN_TEMPLATE' IN TYPEOF(SELF));
      WR4 : (EXISTS(printed_part_template_stack) XOR EXISTS(raw_access_mechanisms)) OR ('AP210_ARM.LAYOUT_MACRO_FLOOR_PLAN_TEMPLATE' IN TYPEOF(SELF));
  END_ENTITY; -- printed_part_template
