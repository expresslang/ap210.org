FUNCTION assigned_product_category_correlation

  FUNCTION assigned_product_category_correlation(
    input1 : SET OF ee_product_version ;
    input2 : SET OF ee_product_related_product_category
  ) : BOOLEAN;

    LOCAL
      d_role_bag : BAG OF STRING;
    END_LOCAL;

    REPEAT i := 1 TO SIZEOF(input2) BY 1;
      d_role_bag := input2[i]\ee_product_category.name + d_role_bag;
    END_REPEAT;
    REPEAT i := 1 TO SIZEOF(d_role_bag) BY 1;
      CASE d_role_bag[i] OF
        'assembly'                  : RETURN(TRUE);
        'bare die'                  : RETURN(TRUE);
        'functionality'             : RETURN(TRUE);
        'interconnect'              : RETURN(TRUE);
        'material'                  : RETURN(TRUE);
        'package'                   : RETURN(TRUE);
        'physical network design'   : RETURN(TRUE);
        'piece part'                : RETURN(TRUE);
        'printed part'              : RETURN(TRUE);
        'requirements model'        : RETURN(TRUE);
        'rule'                      : RETURN(TRUE);
        'simulation model'          : RETURN(TRUE);
        'standard part'             : RETURN(TRUE);
        'technology specific model' : RETURN(TRUE);
        'template model'            : RETURN(TRUE);
        OTHERWISE                   : ;
      END_CASE;
    END_REPEAT;
    RETURN(FALSE);
  END_FUNCTION; -- assigned_product_category_correlation
