ENTITY printed_component

  ENTITY printed_component
    SUPERTYPE OF (ONEOF (layout_macro_component,
 printed_connector_component,
 routed_printed_component))
    SUBTYPE OF (assembly_component);
      cad_layer                                : OPTIONAL external_definition;
      stratum_feature_implementation           : OPTIONAL SET[1:?] OF stratum_feature;
      SELF\assembly_component.usage_definition : printed_part_template_or_printed_part_cross_section_template;
      printed_component_stack                  : OPTIONAL SET[1:?] OF printed_component_link;
      required_material_stack                  : OPTIONAL layer;
    DERIVE
      access_mechanism : SET[1:?] OF printed_component_join_terminal := raw_access_mechanism + printed_component_access_mechanisms(printed_component_stack);
    INVERSE
      raw_access_mechanism       : SET[1:?] OF printed_component_join_terminal FOR associated_component;
      associated_precedent_link  : SET[0:1] OF printed_component_link FOR precedent_printed_component;
      associated_subsequent_link : SET[0:1] OF printed_component_link FOR subsequent_printed_component;
    WHERE
      WR1 : (NOT EXISTS(stratum_feature_implementation) OR same_stratum(stratum_feature_implementation)) OR ('AP210_ARM.LAYOUT_MACRO_COMPONENT' IN TYPEOF(SELF));
      WR2 : (NOT ((EXISTS(SELF\assembly_component.usage_definition.of_stratum_technology)) XOR EXISTS(stratum_feature_implementation))) OR ('AP210_ARM.LAYOUT_MACRO_COMPONENT' IN TYPEOF(SELF));
      WR3 : (NOT ((EXISTS(SELF\assembly_component.usage_definition.material_stack)) XOR EXISTS(required_material_stack))) OR ('AP210_ARM.LAYOUT_MACRO_COMPONENT' IN TYPEOF(SELF));
      WR4 : (NOT ((EXISTS(SELF\assembly_component.usage_definition.printed_part_template_stack)) XOR EXISTS(printed_component_stack))) OR ('AP210_ARM.LAYOUT_MACRO_COMPONENT' IN TYPEOF(SELF));
      WR5 : (NOT (EXISTS(printed_component_stack)) OR consistent_stack_definition(SELF\assembly_component.usage_definition,
 printed_component_stack)) OR ('AP210_ARM.LAYOUT_MACRO_COMPONENT' IN TYPEOF(SELF));
      WR6 : (NOT (EXISTS(associated_precedent_link) OR EXISTS(associated_subsequent_link)) OR EXISTS(stratum_feature_implementation)) OR ('AP210_ARM.LAYOUT_MACRO_COMPONENT' IN TYPEOF(SELF));
      WR7 : NOT ('AP210_ARM.ROUTED_PRINTED_COMPONENT' IN TYPEOF(SELF)) OR ('AP210_ARM.PRINTED_PART_CROSS_SECTION_TEMPLATE' IN TYPEOF(SELF\assembly_component.usage_definition));
      WR8 : NOT ('AP210_ARM.PRINTED_PART_CROSS_SECTION_TEMPLATE' IN TYPEOF(SELF\assembly_component.usage_definition)) OR ('AP210_ARM.ROUTED_PRINTED_COMPONENT' IN TYPEOF(SELF));
  END_ENTITY; -- printed_component
