   FUNCTION msb_shells
      (brep : manifold_solid_brep;
       schema_name : STRING ) : SET [1:?] OF closed_shell;
      IF schema_name + '.BREP_WITH_VOIDS' IN TYPEOF(brep) THEN
         RETURN (brep\brep_with_voids.voids + brep.outer);
      ELSE
         RETURN ([ brep.outer ]);
      END_IF;
   END_FUNCTION;
