   ENTITY assembly_bond_definition
   SUBTYPE OF (shape_aspect);
   UNIQUE
      UR1 : SELF\shape_aspect.name;
   WHERE
      WR1:
         SIZEOF(USEDIN(SELF, 'ELECTRONIC_ASSEMBLY_INTERCONNECT_AND_PACKAGING_DESIGN.' + 'MATERIAL_DESIGNATION.DEFINITIONS')) = 1;
      WR2:
         SIZEOF(QUERY (aga <* USEDIN(SELF, 'ELECTRONIC_ASSEMBLY_INTERCONNECT_AND_PACKAGING_DESIGN.' + 'APPLIED_GROUP_ASSIGNMENT.ITEMS')| ('ELECTRONIC_ASSEMBLY_INTERCONNECT_AND_PACKAGING_DESIGN.' + 'BOND_CATEGORY' IN TYPEOF(aga.assigned_group)))) = 1;
      WR3:
         SIZEOF(QUERY (sar <* USEDIN(SELF, 'ELECTRONIC_ASSEMBLY_INTERCONNECT_AND_PACKAGING_DESIGN.' + 'SHAPE_ASPECT_RELATIONSHIP.RELATING_SHAPE_ASPECT')| (sar\shape_aspect_relationship.name = 'bonded feature 1'))) = 1;
      WR4:
         SIZEOF(QUERY (sar <* USEDIN(SELF, 'ELECTRONIC_ASSEMBLY_INTERCONNECT_AND_PACKAGING_DESIGN.' + 'SHAPE_ASPECT_RELATIONSHIP.RELATING_SHAPE_ASPECT')| (sar\shape_aspect_relationship.name = 'bonded feature 2'))) = 1;
      WR5:
         ('ELECTRONIC_ASSEMBLY_INTERCONNECT_AND_PACKAGING_DESIGN.' + 'EXTERNALLY_DEFINED_PRODUCT_DEFINITION' IN TYPEOF(SELF.of_shape.definition)) AND (SIZEOF(QUERY (prpc <* USEDIN(SELF.of_shape.definition.formation.of_product, 'ELECTRONIC_ASSEMBLY_INTERCONNECT_AND_PACKAGING_DESIGN.' + 'PRODUCT_RELATED_PRODUCT_CATEGORY.' + 'PRODUCTS')| (prpc\product_category.name = 'technology specific model'))) >= 1);
   END_ENTITY;
