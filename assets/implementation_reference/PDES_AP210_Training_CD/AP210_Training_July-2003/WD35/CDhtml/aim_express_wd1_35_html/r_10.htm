RULE approval_requires_approval_person_organization



   RULE approval_requires_approval_person_organization FOR (approval, approval_person_organization );
   WHERE
      WR1:
         SIZEOF(QUERY (app <* approval| NOT (SIZEOF(QUERY (apo <* approval_person_organization| (app :=: apo.authorized_approval))) >= 1))) = 0;
   END_RULE;
