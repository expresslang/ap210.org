      WR1:
         SIZEOF(QUERY (dr <* date_role| NOT (dr\date_role.name IN [ 'creation date', 'request date', 'release date', 'start date', 'contract date', 'certification date', 'sign off date', 'classification date', 'declassification date' ]))) = 0;
   END_RULE;
