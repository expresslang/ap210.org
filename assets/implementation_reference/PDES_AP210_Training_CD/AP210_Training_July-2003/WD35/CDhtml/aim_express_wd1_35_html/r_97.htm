RULE ee_product_definition_constraint



   RULE ee_product_definition_constraint FOR (product_definition_with_associated_documents );
   WHERE
      WR1:
         SIZEOF(QUERY (pdwad <* product_definition_with_associated_documents| NOT (SIZEOF(QUERY (adata <* USEDIN(pdwad, 'ELECTRONIC_ASSEMBLY_INTERCONNECT_AND_PACKAGING_DESIGN.' + 'APPLIED_DATE_AND_TIME_ASSIGNMENT.ITEMS')| (adata.role\date_time_role.name = 'creation date'))) = 1))) = 0;
      WR2:
         SIZEOF(QUERY (pdwad <* product_definition_with_associated_documents| NOT ((SIZEOF(QUERY (aoa <* USEDIN(pdwad, 'ELECTRONIC_ASSEMBLY_INTERCONNECT_AND_PACKAGING_DESIGN.' + 'APPLIED_ORGANIZATION_ASSIGNMENT.ITEMS')| (aoa.role\organization_role.name = 'creator'))) = 1) OR (SIZEOF(QUERY (apaoa <* USEDIN(pdwad, 'ELECTRONIC_ASSEMBLY_INTERCONNECT_AND_PACKAGING_DESIGN.' + 'APPLIED_PERSON_AND_ORGANIZATION_ASSIGNMENT.ITEMS')| (apaoa.role\person_and_organization_role.name = 'creator'))) = 1)))) = 0;
      WR3:
         SIZEOF(QUERY (pdwad <* product_definition_with_associated_documents| NOT (SIZEOF(USEDIN(pdwad, 'ELECTRONIC_ASSEMBLY_INTERCONNECT_AND_PACKAGING_DESIGN.' + 'APPLIED_APPROVAL_ASSIGNMENT.ITEMS')) = 1))) = 0;
      WR4:
         SIZEOF(QUERY (pdwad <* product_definition_with_associated_documents| NOT (SIZEOF(USEDIN(pdwad, 'ELECTRONIC_ASSEMBLY_INTERCONNECT_AND_PACKAGING_DESIGN.' + 'APPLIED_SECURITY_CLASSIFICATION_ASSIGNMENT.ITEMS')) = 1))) = 0;
   END_RULE;
