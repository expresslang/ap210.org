RULE configuration_item_requires_person_organization



   RULE configuration_item_requires_person_organization FOR (configuration_item, applied_person_and_organization_assignment, applied_organization_assignment );
   WHERE
      WR1:
         SIZEOF(QUERY (ci <* configuration_item| NOT ((SIZEOF(QUERY (apaoa <* applied_person_and_organization_assignment| (ci IN apaoa.items))) = 1) OR (SIZEOF(QUERY (aoa <* applied_organization_assignment| (ci IN aoa.items))) = 1)))) = 0;
   END_RULE;
