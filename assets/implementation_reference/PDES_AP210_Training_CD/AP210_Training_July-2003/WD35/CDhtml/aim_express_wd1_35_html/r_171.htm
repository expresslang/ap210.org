RULE positional_boundary_member_definition_constraint



   RULE positional_boundary_member_definition_constraint FOR (property_definition );
   WHERE
      WR1:
         SIZEOF(QUERY (pd <* property_definition| (pd\property_definition.description = 'positional boundary member property') AND NOT ('ELECTRONIC_ASSEMBLY_INTERCONNECT_AND_PACKAGING_DESIGN.' + 'POSITIONAL_BOUNDARY_MEMBER' IN TYPEOF(pd.definition)))) = 0;
      WR2:
         SIZEOF(QUERY (pd <* property_definition| (pd\property_definition.description = 'positional boundary member property') AND NOT (SIZEOF(QUERY (pdr <* USEDIN(pd, 'ELECTRONIC_ASSEMBLY_INTERCONNECT_AND_PACKAGING_DESIGN.' + 'PROPERTY_DEFINITION_RELATIONSHIP.RELATING_PROPERTY_DEFINITION')| ('ELECTRONIC_ASSEMBLY_INTERCONNECT_AND_PACKAGING_DESIGN.' + 'DIMENSIONAL_SIZE_PROPERTY' IN TYPEOF(pdr.related_property_definition)) AND (pdr\property_definition_relationship.name = 'boundary member associated dimension'))) = 1))) = 0;
   END_RULE;
