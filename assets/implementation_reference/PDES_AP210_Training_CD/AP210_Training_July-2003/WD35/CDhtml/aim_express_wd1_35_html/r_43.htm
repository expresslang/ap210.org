RULE connection_zone_constraint



   RULE connection_zone_constraint FOR (shape_aspect, representation );
   WHERE
      WR1:
         SIZEOF(QUERY (sa <* shape_aspect| (sa\shape_aspect.description = 'connection zone') AND NOT (SIZEOF(USEDIN(sa, '')) >= 1))) = 0;
      WR2:
         SIZEOF(QUERY (sa <* shape_aspect| (sa\shape_aspect.description = 'connection zone') AND NOT (SIZEOF(QUERY (sar <* USEDIN(sa, 'ELECTRONIC_ASSEMBLY_INTERCONNECT_AND_PACKAGING_DESIGN.' + 'SHAPE_ASPECT_RELATIONSHIP.RELATED_SHAPE_ASPECT')| (sar\shape_aspect_relationship.name IN [ 'connection', 'minimum attachment region size', 'maximum attachment region size', 'external connection area', 'external connection zone', 'internal connection zone', 'mounting area', 'terminal connection zone' ]))) >= 1))) = 0;
   END_RULE;
