RULE geometric_tolerance_qualifier_constraint



   RULE geometric_tolerance_qualifier_constraint FOR (measure_qualification );
   WHERE
      WR1:
         SIZEOF(QUERY (mq <* measure_qualification| ((mq\measure_qualification.name = 'predefined geometric tolerance qualifier') OR (mq\measure_qualification.name = 'user defined geometric tolerance qualifier')) AND NOT (('ELECTRONIC_ASSEMBLY_INTERCONNECT_AND_PACKAGING_DESIGN.' + 'MEASURE_WITH_UNIT' IN TYPEOF(mq.qualified_measure)) AND (SIZEOF(QUERY (pugt <* USEDIN(mq.qualified_measure, 'ELECTRONIC_ASSEMBLY_INTERCONNECT_AND_PACKAGING_DESIGN.' + 'GEOMETRIC_TOLERANCE.MAGNITUDE')| ('ELECTRONIC_ASSEMBLY_INTERCONNECT_AND_PACKAGING_DESIGN.' + 'PHYSICAL_UNIT_GEOMETRIC_TOLERANCE' IN TYPEOF(pugt)))) = 0)))) = 0;
   END_RULE;
