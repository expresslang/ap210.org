RULE versioned_action_request_requires_date_or_date_and_time



   RULE versioned_action_request_requires_date_or_date_and_time FOR (versioned_action_request, applied_date_and_time_assignment, applied_date_assignment );
   WHERE
      WR1:
         SIZEOF(QUERY (varq <* versioned_action_request| NOT ((SIZEOF(QUERY (adata <* applied_date_and_time_assignment| (varq IN adata.items))) = 1) OR (SIZEOF(QUERY (ada <* applied_date_assignment| (varq IN ada.items))) = 1)))) = 0;
   END_RULE;
