   ENTITY colour_rgb
   SUBTYPE OF (colour_specification);
      red : REAL;
      green : REAL;
      blue : REAL;
   WHERE
      WR1:
         (0.00000 <= red) AND (red <= 1.00000);
      WR2:
         (0.00000 <= green) AND (green <= 1.00000);
      WR3:
         (0.00000 <= blue) AND (blue <= 1.00000);
   END_ENTITY;
