RULE restraint_condition_constraint



   RULE restraint_condition_constraint FOR (property_definition );
   WHERE
      WR1:
         SIZEOF(QUERY (pd <* property_definition| ((pd\property_definition.description = 'restraint') OR (pd\property_definition.description = 'tolerance specific restraint')) AND (SIZEOF(QUERY (pdr <* USEDIN(pd, 'ELECTRONIC_ASSEMBLY_INTERCONNECT_AND_PACKAGING_DESIGN.' + 'PROPERTY_DEFINITION_RELATIONSHIP.RELATING_PROPERTY_DEFINITION')| ('ELECTRONIC_ASSEMBLY_INTERCONNECT_AND_PACKAGING_DESIGN.' + 'REQUIREMENTS_PROPERTY' IN TYPEOF(pdr.related_property_definition)) AND (pdr\property_definition_relationship.name = 'restraint description'))) = 0))) = 0;
   END_RULE;
