RULE certification_requires_date_or_date_and_time



   RULE certification_requires_date_or_date_and_time FOR (certification, applied_date_and_time_assignment, applied_date_assignment );
   WHERE
      WR1:
         SIZEOF(QUERY (cert <* certification| NOT ((SIZEOF(QUERY (adata <* applied_date_and_time_assignment| (cert IN adata.items))) = 1) OR (SIZEOF(QUERY (ada <* applied_date_assignment| (cert IN ada.items))) = 1)))) = 0;
   END_RULE;
