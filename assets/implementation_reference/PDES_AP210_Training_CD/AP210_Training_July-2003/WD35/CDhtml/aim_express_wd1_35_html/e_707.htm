   ENTITY statistical_geometric_tolerance
   SUBTYPE OF (physical_unit_geometric_tolerance);
   WHERE
      WR1:
         SIZEOF(QUERY (gtr <* USEDIN(SELF, 'ELECTRONIC_ASSEMBLY_INTERCONNECT_AND_PACKAGING_DESIGN.' + 'GEOMETRIC_TOLERANCE_RELATIONSHIP.RELATED_GEOMETRIC_TOLERANCE')| ('ELECTRONIC_ASSEMBLY_INTERCONNECT_AND_PACKAGING_DESIGN.' + 'PHYSICAL_UNIT_GEOMETRIC_TOLERANCE' IN TYPEOF(gtr.relating_geometric_tolerance)) AND (gtr\geometric_tolerance_relationship.name = 'substitutable geometric tolerance'))) <= 1;
      WR2:
         SIZEOF(QUERY (pdr <* USEDIN(SELF, 'ELECTRONIC_ASSEMBLY_INTERCONNECT_AND_PACKAGING_DESIGN.' + 'PROPERTY_DEFINITION_RELATIONSHIP.RELATING_PROPERTY_DEFINITION')| ('ELECTRONIC_ASSEMBLY_INTERCONNECT_AND_PACKAGING_DESIGN.' + 'REQUIREMENTS_PROPERTY' IN TYPEOF(pdr.related_property_definition)) AND (pdr\property_definition_relationship.name = 'geometric tolerance statistical control requirement'))) = 1;
      WR3:
         SIZEOF(QUERY (gtr <* USEDIN(SELF, 'ELECTRONIC_ASSEMBLY_INTERCONNECT_AND_PACKAGING_DESIGN.' + 'GEOMETRIC_TOLERANCE_RELATIONSHIP.RELATED_GEOMETRIC_TOLERANCE')| ('ELECTRONIC_ASSEMBLY_INTERCONNECT_AND_PACKAGING_DESIGN.' + 'STATISTICAL_GEOMETRIC_TOLERANCE' IN TYPEOF(gtr.relating_geometric_tolerance)) AND (gtr\geometric_tolerance_relationship.name = 'substitutable geometric tolerance'))) <= 0;
   END_ENTITY;
