   ENTITY cable_component
   SUBTYPE OF (component_definition);
   WHERE
      WR1:
         SIZEOF(QUERY (ip <* QUERY (pdr <* USEDIN(SELF, 'ELECTRONIC_ASSEMBLY_INTERCONNECT_AND_PACKAGING_DESIGN.' + 'PRODUCT_DEFINITION_RELATIONSHIP.RELATED_PRODUCT_DEFINITION')| (pdr\product_definition_relationship.name = 'instantiated part'))| (SIZEOF([ ('ELECTRONIC_ASSEMBLY_INTERCONNECT_AND_PACKAGING_DESIGN.' + 'PHYSICAL_UNIT') ] * TYPEOF(ip.relating_product_definition)) = 1) AND (ip.relating_product_definition.frame_of_reference.name = 'physical design usage'))) = 1;
      WR2:
         SELF\product_definition.description :<>: 'laminate component';
   END_ENTITY;
